		\item \( y = C e^{2x} \) % \( y' = 2y \)
		\item \( y = Cx \) % \( y' = \frac{y}{x} \)
		\item \( y = Ce^{x} - 1  \) % \( xy' = y + 1 \)
		\item \( y^2 = x^2 + C \) % \( y' = \frac{x}{y} \)
		\item \( \frac{1}{y^2} + x^2 = C \) % \( y' = xy^3 \)
		\item \( e^y + \cos{x} = C \) % \( y' = e^{-y} \sin{x} \)
		\item \( \frac{1}{y} = \ln{\cos{x}} + C \) % \( y' = y^2 \tan{x} \)
		\item \( \sin{y} = Ce^{2x} \) % \( y' = 2\tan{y} \)
		\item \( y \ln{y} - x \ln{x} = y - x + C \) % \( y' \ln{y} = \ln{x} \)
		\item \( y = C^{e^{x^2}} \) % \( y' = 2x y \ln{y} \)
		\item \( \artanh{y} = \arctan{x} + C \) % \( y' = \frac{1 - y^2}{1 + x^2} \)
		\item \( \arctan{y} = \ln{\ln{x}} + C \) % \( y' = \frac{1 + y^2}{x \ln{x}} \)
		\item \( \cot{y} + \tan{x} = C \) % \( y' \cos^2{x} = \sin^2{y} \)
		\item \( \ln{y} + \sqrt{1 - x^2} = C \) % \( y' \sqrt{1 - x^2} = xy \)
		\item \( y = Ce^{x^2 + x} + 1 \) % \( y' = 2xy + y - 2x - 1 \)
		\item \( \pares{y + 1} \cdot e^{-y} + \pares{x - 1} \cdot e^{x} = C \) % \( yy' = xe^{x + y} \)

		\item \( y = C \cdot \cosh{x} \) % \( y' = y \tanh{x} \)
		\item \( \sec{y} + \tan{y} = Cx \) % \( y' = \frac{\cos{y}}{x} \)
		\item \( y = C \cdot \pares{1 + e^{x}} \) % \( \pares{1 + e^x} \cdot y' = y e^{x} \)
		\item \( e^{2y} = \pares{x^2 - 1} \cdot e^{x^2} + C \) % \( y' = x^3 e^{x^2 - 2y} \)
		\item \( \pares{y^2 - 1} \sin{y} + 2x \cos{y} = x^4 + C \) % \( y' = \frac{4x^3 \sec{y}}{y^2 + 1} \)

	\label{sol:firstorder:separable_cauchy}
	\begin{enumsols}
		\item \( y = 1 \) % \( y' = 0, ~ y(0) = 1 \)
		\item \( y = \pares{x + 2}^2 \) % \( y' - 4 = 2x, ~ y(1) = 9 \)
		\item \( y = x \) % \( y' = \frac{y}{x}, ~ y(1) = 1 \)
		\item \( x^2 + y^2 = 1 \) % \( y' = -\frac{x}{y}, ~ y(0) = 1 \)
		\item \( y = 2e^{x^2} \) % \( y' = 2xy, ~ y(0) = 2 \)
		\item \( y = e^{x^2 + x} - 1 \) % \( y' - 1 = 2xy + 2x + y, ~ y(-1) = 0 \)
		\item \( \sqrt{y} = x + 1 \) % \( y' = 2\sqrt{y}, ~ y(0) = 1 \)
		\item \( e^{y} = e^{x} + 1 \) % \( y' = e^{x - y}, ~ y(0) = \ln{2} \)
		\item \( \pares{2y^2 - 2y + 1} \cdot e^{2y} = \pares{x - 1} \cdot e^{x} + 2 \) % \( 4y^2 y' = xe^{x - 2y}, ~ y(0) = 0 \)
		\item \( \sin{y} = e^{x} \) % \( y' = \tan{y}, ~ y(0) = \frac{\pi}{2} \)
		\item \( y = 2\sin{x} \) % \( y' = y \cot{x}, ~ y \pares{\frac{\pi}{2}} = 2 \)
		\item \( y = -3 \pares{\sec{x} + \tan{x}} \) % \( y' \cos{x} = y, ~ y(0) = 3 \)
		\item \( 1 + y^2 = x^2 \) % \( xyy' = 1 + y^2, ~ y(1) = 0 \)
		\item \( y \ln{y} - x \ln{x} = y - x + 1 \) % \( y'\ln{y} = \ln{x}, ~ y(1) = 0 \)
		\item \( y \sin{y} + \cos{y} = x^2 - 1 \) % \( yy' = 2x \sec{y}, ~ y(0) = \pi \)
		\item \( y = \arcsin{x} \) % \( y' \sqrt{1 - x^2} = \frac{y}{\arcsin{x}}, ~ y(1) = \frac{\pi}{2} 
		\item \( \arcosh{y} = \arsinh{x} \) % \( y' \sqrt{1 + x^2} = \sqrt{y^2 - 1}, ~ y(0) = 1 \)
		\item \( \tan{y} = e^{-x} - x^2 \) % \( y' = \pares{2x + e^{-x}} \cdot \cos^2{y}, ~ y(0) = 1 \)
		\item \( 2\sqrt{y} - 2 \ln\pares{\sqrt{y} + 1} = x \) % \( y' = \sqrt{y} + 1, ~ y(1) = 2 - \ln{4} \)
		\item \( \cosec{y} + \cot{y} = Ce^{\arctan{e^x}} \) % \( y' \cosh{x} = \sin{y}, ~ x \to -\infty, ~ y = \frac{\pi}{2} \)
	\end{enumsols}
