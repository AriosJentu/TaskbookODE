\subsection{Задачи}
	
	Для следующих уравнений построить общее решение:
	\begin{multicols}{3}
		\begin{enumtasks}

			\label{firstorder:separable}
			\item \( y' = 2x + 1 \)															% y = x^2 + x + C
			\item \( y' = 2y \)																% y = C e^{2x}
			\item \( y' = \frac{y}{x} \)													% y = Cx
			\item \( xy' = y + 1 \)															% y = Ce^{x} - 1 
			\item \( y' = \frac{x}{y} \)													% y^2 = x^2 + C
			\item \( y' = xy^3 \)															% \frac{1}{y^2} + x^2 = C
			\item \( y' = e^{-y} \sin{x} \)													% e^y + \cos{x} = C
			\item \( y' = y^2 \tan{x} \)													% \frac{1}{y} = \ln{\cos{x}} + C
			\item \( y' = 2\tan{y} \)														% \sin{y} = Ce^{2x}
			\item \( y' \ln{y} = \ln{x} \)													% y \ln{y} - x \ln{x} = y - x + C
			\item \( y' = 2x y \ln{y} \)													% y = C^{e^{x^2}}
			\item \( y' = \frac{1 - y^2}{1 + x^2} \)										% \artanh{y} = \arctan{x} + C
			\item \( y' = \frac{1 + y^2}{x \ln{x}} \)										% \arctan{y} = \ln{\ln{x}} + C
			\item \( y' \cos^2{x} = \sin^2{y} \)											% \cot{y} + \tan{x} = C
			\item \( y' \sqrt{1 - x^2} = xy \)												% \ln{y} + \sqrt{1 - x^2} = C
			\item \( y' = 2xy + y - 2x - 1 \)												% y = Ce^{x^2 + x} + 1
			\item \( yy' = xe^{x + y} \)													% \pares{y + 1} \cdot e^{-y} + \pares{x - 1} \cdot e^{x} = C
			\item \( xy' = 2y\ln{x} \)														% \ln{y} = \ln^2{x} + C
			\item \( y' = y \tanh{x} \)														% y = C \cdot \cosh{x}
			\item \( y' = \frac{\cos{y}}{x} \)												% \sec{y} + \tan{y} = Cx
			\item \( \pares{1 + e^x} \cdot y' = y e^{x} \) 									% y = C \cdot \pares{1 + e^{x}}
			\item \( y' = x^3 e^{x^2 - 2y} \)												% e^{2y} = \pares{x^2 - 1} \cdot e^{x^2} + C
			\item \( y' = \frac{4x^3 \sec{y}}{y^2 + 1} \)									% \pares{y^2 - 1} \sin{y} + 2x \cos{y} = x^4 + C
			\item \( yy' = 2x^3 \cos^2{y^2} \)												% \tan{y^2} = x^4 + C
			\itemstar \( y' \cdot \cosh{x} = \sec{y} \)										% \sin{y} = 2 \arctan{e^{x}} + C
			\itemstar \( y^{\ln{y}} \cdot y' = \frac{x^{\frac{1}{\ln{x}}}}{2\ln{y} + 1} \)	% y^{\ln{y} + 1} = xe + C
			\itemstar \( \pares{x-1}^4 \cdot y' = x^4 \cdot e^{-y} \)						% \pares{x-1}^3 \cdot e^{y} = 4\ln\pares{x - 1} \cdot \pares{x-1}^3 + x^4 - 3x^3 - 3x^2 + 9x + C
			
		\end{enumtasks}
	\end{multicols}

	\begin{enumtasks}

		\label{firstorder:separable2}
		\itemstar \( y' \cdot \arcsin{y} = \sqrt{1 - y^2 - x^2 + x^2 y^2} \)			% \arcsin^2{y} = x \sqrt{1 - x^2} + \arcsin{x} + C
		\itemstar \( 2yy' \cos{x} + y \sin{x} - y^2 = y^2 \sin{x} + y' \cos{x} - y \)	% y^2 - y = C \cdot \sec{x} \cdot \pares{\sec{x} + \tan{x}}
		\itemstar \( 3y' = \cot^4{y} \cdot \sqrt{2\tan{x}} \)							% \tan^3{y} - 3\tan{y} + 3y = \arctan{\frac{\tan{x} - 1}{\sqrt{2 \tan{x}}}} - \artanh{\frac{\tan{x} + 1}{\sqrt{2 \tan{x}}}} + C

	\end{enumtasks}

	Найти решение следующих уравнений, удовлетворяющее начальным условиям:
	\begin{multicols}{2}
		\begin{enumtasks}

			\label{firstorder:separable_cauchy}
			\item \( y' = 0, ~ y(0) = 1 \) 													% y = 1
			\item \( y' - 4 = 2x, ~ y(1) = 9 \) 											% y = \pares{x + 2}^2
			\item \( y' = \frac{y}{x}, ~ y(1) = 1 \) 										% y = x
			\item \( y' = -\frac{x}{y}, ~ y(0) = 1 \) 										% x^2 + y^2 = 1
			\item \( y' = 2xy, ~ y(0) = 2 \)												% y = 2e^{x^2}
			\item \( y' - 1 = 2xy + 2x + y, ~ y(-1) = 0 \) 									% y = e^{x^2 + x} - 1
			\item \( y' = 2\sqrt{y}, ~ y(0) = 1 \)											% \sqrt{y} = x + 1
			\item \( y' = e^{x - y}, ~ y(0) = \ln{2} \)										% e^{y} = e^{x} + 1
			\item \( 4y^2 y' = xe^{x - 2y}, ~ y(0) = 0 \)									% \pares{2y^2 - 2y + 1} \cdot e^{2y} = \pares{x - 1} \cdot e^{x} + 2
			\item \( y' = \tan{y}, ~ y(0) = \frac{\pi}{2} \)								% \sin{y} = e^{x}
			\item \( y' = y \cot{x}, ~ y \pares{\frac{\pi}{2}} = 2 \)						% y = 2\sin{x}
			\item \( y' \cos{x} = y, ~ y(0) = 3 \) 											% y = -3 \pares{\sec{x} + \tan{x}}
			\item \( xyy' = 1 + y^2, ~ y(1) = 0 \)											% 1 + y^2 = x^2
			\item \( y'\ln{y} = \ln{x}, ~ y(1) = 0 \) 										% y \ln{y} - x \ln{x} = y - x + 1
			\item \( yy' = 2x \sec{y}, ~ y(0) = \pi \)										% y \sin{y} + \cos{y} = x^2 - 1
			\item \( y' \sqrt{1 - x^2} = \frac{y}{\arcsin{x}}, ~ y(1) = \frac{\pi}{2} \)	% y = \arcsin{x}
			\item \( y' \sqrt{1 + x^2} = \sqrt{y^2 - 1}, ~ y(0) = 1 \) 						% \arcosh{y} = \arsinh{x}
			\item \( y' = \pares{2x + e^{-x}} \cdot \cos^2{y}, ~ y(0) = 1 \)				% \tan{y} = e^{-x} - x^2
			\item \( y' = \sqrt{y} + 1, ~ y(1) = 2 - \ln{4} \) 								% 2\sqrt{y} - 2 \ln\pares{\sqrt{y} + 1} = x
			\item \( y' \cosh{x} = \sin{y}, ~ x \to -\infty, ~ y = \frac{\pi}{2} \) 		% \cosec{y} + \cot{y} = Ce^{\arctan{e^x}}

		\end{enumtasks}
	\end{multicols}
